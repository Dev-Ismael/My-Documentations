    (3) Routes 
    --------------------------------------
        ==> Routes is URL in laravel , there is many types in routes 
            Route::get($uri, $callback);
            Route::post($uri, $callback);
            Route::put($uri, $callback);
            Route::patch($uri, $callback);
            Route::delete($uri, $callback);
            Route::options($uri, $callback);

        ==> Sometimes you may need to register a route that responds to multiple HTTP verbs
            Route::match(['get', 'post'], '/', function () {
                //
            });
            Route::any('/', function () {
                //
            });

        ==> Route::get($uri, $callback); use to normal move with links
        Route::get('/ahmed', function () {
            return view("ahmed") ;
        });

        ==> Route::post($uri, $callback); use to recieved data from forms 
        Route::post('/ali', function () {
            return view('ali');
        });
        any page go to ali.blade.php must have form to direct to it

        ==> Route::put($uri, $callback); use to update data 
        <form action="/foo/bar" method="POST">
            <input type="hidden" name="_method" value="PUT">
            <input type="text" name="name" placeholder="username">
            <input type="submit" value="Click Here">
        </form>
        OR
        <form action="/foo/bar" method="POST">
            @method('PUT')
            <input type="text" name="name" placeholder="username">
            <input type="submit" value="Click Here">
        </form>

        ==> Route::delete($uri, $callback); use to delete data and usage the same put 

        ==> Route::patch($uri, $callback); use to delete data and usage the same put 

        ==> Route::any($uri, $callback); use to be ( post , patch , delete ) 

        ==> we can set name for route for use it in blade file 

        Ex ==> 
        Route::get('users/{id}', function () {
            return view("users") ;
        }) -> name("a");

        and in blade file 

        <a href="{{ route("a" , 55 ) }}"> Click Here </a>


        ==> 
        // resource mean all crud methods ( index , show ,create , store , edit , update , destroy )
        Route::resource("posts","Postcontroller")->except("create" , "edit" , "show");
        // use this command to know all routes ==>
        ==> php artisan route:list

    (5) regular expression with routes   
    --------------------------------------

    Route::get('user/{id?}', function ( $ident = null ) { 
        return "hello user number " . $ident ;
    })->where("id" , "[0-9]+");

    Route::get('user/{id?}/{username}', function ( ) { 
        return "hello user number " ;
    })->where( "id" => "[0-9]+" , "username" => "[A-Za-z]+" );

    OR we can apply all regular expression with routes with 

    Route::pattern("id" , "[0-9]+");
    Route::pattern( "id" => "[0-9]+" , "username" => "[a-z]+" );

    (13) passing data from controller to view     
    --------------------------------------
        1 - passing data with [ with() function ] 
        ---------------------------------------------
            ==> in controller file 
            Route::get('/', function () {
                return view('home')->with("name" , "AbdullRahman");
                OR                
                return view('home')->with( [ "name" => "AbdullRahman" , "age" => "23" ] );
            });
            ==> in blade file 
            <h3> {{ $name }} </h3>
        

        2- passing data as array 
        ---------------------------
            ==> in controller file 
            Route::get('/', function () {
                $data = [] ;
                $data["name"] = "AbdullRahman";
                $data["age"] = "23";
                OR 
                $data = [ "name" => "AbdullRahman" , "age" => "23" ]
                return view('home' , $data );
            });
            ==> in blade file 
            <h3> {{ $name }} </h3>
            <h3> {{ $age }} </h3>


        3- passing data as object 
        -------------------------------
            ==> in controller file 
            Route::get('/', function () {

                $obj = new \stdClass();
                $obj-> name = "AbdullRahman";
                $obj-> age = "23";
                return view('home' , compact("obj") );
                
            });
            ==> in blade file 
            <h3> {{ $obj -> name }} </h3>
            <h3> {{ $obj -> age }} </h3>
